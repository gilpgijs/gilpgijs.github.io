Atributos

<!DOCTYPE html>
<!--
  Los elementos tienen preestablecidas características
  llamadas atributos.
  Cada atributo tiene asociado un nombre y un valor.
  Los atributos se colocan en una etiqueta después del
  tipo de elemento.
  Los atributos tienen valores predeterminados que se
  pueden modificar en el código.
-->
<!--El atributo lang tiene el valor es -->
<html lang="es">
  <head>
    <!--El atributo charset tiene el valor UTF-8 -->
    <meta charset="UTF-8">
    <!--
      El atributo name tiene el valor viewport y el
      atributo content tiene el valor
      width=device-width, initial-scale=1.0
    -->
    <meta name="viewport"
      content="width=device-width, initial-scale=1.0">
    <title>Hola Mundo</title>
  </head>
  <body>
    <h1>Hola Mundo</h1>
    <p>Hola mundo.</p>
    <p>Este es mi primer ejemplo.</p>
    <footer>
      <p>&copy; 2020 Gilberto Pacheco Gallegos.</p>
    </footer>
  </body>
</html>
