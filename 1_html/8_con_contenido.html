Elementos con Contenido

<!DOCTYPE html>
<!--
  Algunos elementos llevan tipo, opcionalmente
  atributos y contenido (que incluye texto y
  etiquetas).
  Se utilizan 2 etiquetas:

  ETIQUETA INICIAL
  Inicia el elemento y se colocan el tipo y
  opcionalmente los atributos.

  ETIQUETA FINAL O DE CIERRE
  Termina la definición del elemento.
  Inicia con / y luego el tipo del elemento que
  cierran.

  El contenido se coloca entre la etiquetas iniciales
  y finales.
-->
<!-- html es un tipo de elemento con contenido. -->
<html lang="es">
  <!-- head es un tipo de elemento con contenido. -->
  <head><!-- En esta línea inicia head. -->
    <meta charset="UTF-8">
    <meta name="viewport"
      content="width=device-width, initial-scale=1.0">
    <!--
      title es un tipo de elemento con contenido.
    -->
    <title>Hola Mundo</title>
  </head><!-- En esta línea termina head. -->
  <!-- head es un tipo de elemento con contenido. -->
  <body><!-- En esta línea inicia body. -->
    <!-- h1 es un tipo de elemento con contenido. -->
    <h1>Hola Mundo</h1>
    <!-- p es un tipo de elemento con contenido. -->
    <p>Hola mundo.</p>
    <p>Este es mi primer ejemplo.</p>
    <!--
      footer es un tipo de elemento con contenido.
    -->
    <footer><!-- En esta línea inicia footer. -->
      <p>&copy; 2020 Gilberto Pacheco Gallegos.</p>
    </footer><!-- En esta línea termina footer. -->
  </body><!-- En esta línea termina body. -->
</html>
