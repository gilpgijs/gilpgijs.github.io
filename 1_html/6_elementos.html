Elementos

<!DOCTYPE html>
<!--
  Un elemento es un objeto que representa una forma de
  procesar el texto y se escribe como una etiqueta.
  Cada elemento tiene asociado un tipo, que representa
  su forma de procesamiento.
  Cada tipo de elemento tiene un funcionamiento
  preestablecido por el navegador.
  El tipo es la primera palabra de la etiqueta.
-->
<html lang="es"><!--Este elemento es de tipo html.-->
  <head><!-- Este elemento es de tipo head. -->
    <!-- Este elemento es de tipo meta. -->
    <meta charset="UTF-8">
    <!-- Este elemento es de tipo meta. -->
    <meta name="viewport"
      content="width=device-width, initial-scale=1.0">
    <!-- Este elemento es de tipo title. -->
    <title>Hola Mundo</title>
  </head>
  <body><!-- Este elemento es de tipo body. -->
    <!-- Este elemento es de tipo h1. -->
    <h1>Hola Mundo</h1>
    <!-- Este elemento es de tipo p. -->
    <p>Hola mundo.</p>
    <!-- Este elemento es de tipo p. -->
    <p>Este es mi primer ejemplo.</p>
    <!-- Este elemento es de tipo footer. -->
    <footer>
      <!-- Este elemento es de tipo p. -->
      <p>&copy; 2020 Gilberto Pacheco Gallegos.</p>
    </footer>
  </body>
</html>
