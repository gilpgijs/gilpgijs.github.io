El Elemento script

El elemento script contiene las instrucciones de JavaScript.

Su especificación está en https://html.spec.whatwg.org/multipage/scripting.html#scripting-3

<!DOCTYPE html>
<html lang="es">
  <head>
    <meta charset="UTF-8">
    <meta name="viewport"
      content="width=device-width, initial-scale=1.0">
    <title>Saludo</title>
  </head>
  <body>
    <form action="javascript:saluda();">
      <h1>Saludo</h1>
      <p>
        <label>
          Nombre
          <input id="inNombre" type="text" required>
        </label>
      </p>
      <p><button>Calcular</button></p>
      <p><output id=outSaludo></output></p>
    </form>
    <footer>
      <p>&copy; 2020 Gilberto Pacheco Gallegos.</p>
    </footer>
    <!--
      El elemento script tiene el código JavaScript
      de la página.
    -->
    <script>
      /* Este código no se ejecuta al inicio; es
       * necesario que se ejecute la instrucción
       * saluda();
       * y con ello se transfiere la ejecución a
       * esta function. */
      function saluda() {
        const nombre = inNombre.value;
        const saludo = `Hola ${nombre}.`;
        outSaludo.value = saludo;
      }
    </script> <!-- Fin del script. -->
  </body>
</html>
